$\begin{array}{l} \left\{a_{i}, a_{j}^\dagger\right\}=\delta_{i j} \quad\left\{a_{i}, a_{j}\right\}=\left\{a_{i}^{\dagger}, a_{j}^{\dagger}\right\}=0 \\ a_{i}\left|n_{i}\right\rangle=\sqrt{n_{i}}\left|n_{i}-1\right\rangle \\ a_{i}{ }^{\dagger}\left|n_{i}\right\rangle=\sqrt{1-n_{i}}\left|n_{i}+1\right\rangle \\ a_{i}\left|n_{1} n_{2} \cdots n_{i} \cdots\right\rangle=(-1)^{m} \sqrt{n_{i}}\left|n_{1} n_{2} \cdots n_{i}-1 \cdots\right\rangle \\ a_{i}^{\dagger}\left|n_{1} n_{2} \cdots n_{i} \cdots\right\rangle=(-1)^{m} \sqrt{1-n_{i}}\left|n_{1} n_{2}\cdots n_{i}+1 \cdots\right\rangle \\ m \equiv \sum_{\alpha=1}^{i-1} n_{\alpha} \end{array}$

对boson体系有：

$\begin{array}{l} [a_i, a_{j}^\dagger]=\delta_{i j} \quad[a_{i}, a_{j}]=[a_{i}^{\dagger}, a_{j}^{\dagger}]=0 \\ a_{i}\left|n_{i}\right\rangle=\sqrt{n_{i}}\left|n_{i}-1\right\rangle \\ a_{i}{ }^{\dagger}\left|n_{i}\right\rangle=\sqrt{1+n_{i}}\left|n_{i}+1\right\rangle \\ a_{i}\left|n_{1} n_{2} \cdots n_{i} \cdots\right\rangle=\sqrt{n_{i}}\left|n_{1} n_{2} \cdots n_{i}-1 \cdots\right\rangle \\ a_{i}^{\dagger}\left|n_{1} n_{2} \cdots n_{i} \cdots\right\rangle= \sqrt{1+n_{i}}\left|n_{1} n_{2}\cdots n_{i}+1 \cdots\right\rangle \\ \end{array}$
